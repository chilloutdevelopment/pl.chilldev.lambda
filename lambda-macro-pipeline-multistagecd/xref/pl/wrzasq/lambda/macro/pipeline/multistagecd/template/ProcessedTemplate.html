73      /**
74       * Builds pipeline resource.
75       *
76       * @param input Initial template structure.
77       * @return Template state after processing.
78       */
79      private Map<String, Object> processTemplate(Map<String, Object> input) {
80          logger.info("Found pipeline definition.");
81          // default values are set in the classes
82          var pipeline = ProcessedTemplate.objectMapper.convertValue(input.get(ProcessedTemplate.SECTION_PIPELINE), PipelineDefinition.class);
83          input.remove(ProcessedTemplate.SECTION_PIPELINE);
84          var output = new HashMap<>(input);
85          // pipeline metadata setup
86          output.put(TemplateUtils.SECTION_CONDITIONS, this.buildConditions(TemplateUtils.asMap(input.computeIfAbsent(TemplateUtils.SECTION_CONDITIONS, key -> new HashMap<>())), pipeline.getConfig()));
87          output.put(TemplateUtils.SECTION_RESOURCES, this.buildResources(TemplateUtils.asMap(input.computeIfAbsent(TemplateUtils.SECTION_RESOURCES, key -> new HashMap<>())), pipeline));
88          return output;
89      }

98      private Map<String, Object> buildConditions(Map<String, Object> conditions, PipelineConfiguraiton config) {
99          conditions.put(config.getHasCheckoutStepConditionName(), ProcessedTemplate.buildBooleanCondition(config.getHasCheckoutStepParameterName()));
100         conditions.put(config.getHasNextStageConditionName(), ProcessedTemplate.buildBooleanCondition(config.getHasNextStageParameterName()));
101         conditions.put(config.getRequiresManualApprovalConditionName(), Collections.singletonMap("Fn::And", Arrays.asList(Collections.singletonMap("Condition", config.getHasNextStageConditionName()), ProcessedTemplate.buildBooleanCondition(config.getRequiresManualApprovalParameterName()))));
102         return conditions;
103     }

114         resources.put(config.getResourceName(), this.buildPipelineDefinition(pipeline));
115         // GitHub webhook
116         if (config.getWebhookAuthenticationType() != null && config.getWebhookSecretToken() != null) {
117             resources.put(String.format("%sWebhook", config.getResourceName()), this.buildWebhookDefinition(config, pipeline.getSources().keySet().iterator().next()));

122     /**
123      * Build pipeline definition.
124      *
125      * @param pipeline Pipeline definition.
126      * @return Resource definition.
127      */
128     private Map<String, Object> buildPipelineDefinition(PipelineDefinition pipeline) {
129         var config = pipeline.getConfig();
130         var artifacts = pipeline.getArtifacts();
131         var stages = new ArrayList<>();
132         stages.add(this.buildSourceStage(config, pipeline.getSources(), artifacts));
133         stages.addAll(this.buildStages(pipeline.getStages()));
134         stages.add(this.buildManualApprovalStage(config));
135         stages.add(this.buildPromoteStage(config, artifacts));
136         // finalize properties
137         var properties = new HashMap<>(pipeline.getProperties());
138         properties.put("Stages", stages);
139         return TemplateUtils.generateResource("CodePipeline::Pipeline", properties, null);
140     }

154     /**
155      * Builds sources action steps.
156      *
157      * @param sources Source locations.
158      * @return Stage actions.
159      */
160     private Object buildCheckoutSources(Map<String, Map<String, Object>> sources) {
161         var steps = new ArrayList<>(sources.size());
162         // keeping order to avoid structure changes in CloudFormation
163         for (var id : new TreeSet<>(sources.keySet())) {
164             steps.add(ProcessedTemplate.populateActionDefinition(sources.get(id), id, null, Collections.emptyMap(), Collections.emptyList(), Collections.singletonList(id), null));
165         }
166         return steps;
167     }

169     /**
170      * Builds sources action steps.
171      *
172      * @param artifacts Previous stage artifacts.
173      * @return Stage actions.
174      */
175     private Object buildPromotedSources(Map<String, PipelineArtifact> artifacts) {
176         var steps = new ArrayList<>(artifacts.size());
177         var runOrder = 0;
178         /*
179         having sources in reverse direction than in promote step, makes us sure that each next stage pipeline is
180         launched only when all previous stage artifacts are properly uploaded
181          */
182         for (var id : new TreeSet<>(artifacts.keySet()).descendingSet()) {
183             var artifact = artifacts.get(id);
184             var type = ActionTypeId.s3Source();
185             var actionConfig = new HashMap<String, Object>();
186             actionConfig.put("S3Bucket", artifact.getSourceBucketName());
187             actionConfig.put("S3ObjectKey", artifact.getObjectKey());
188             steps.add(ProcessedTemplate.buildActionDefinition(id, type, actionConfig, Collections.emptyList(), Collections.singletonList(id), ++runOrder));
189         }
190         return steps;
191     }

213     /**
214      * Builds artifacts promotion stage definition.
215      *
216      * @param config Pipeline configuration.
217      * @param artifacts Artifacts configuration.
218      * @return Stage definition.
219      */
220     private Object buildPromoteStage(PipelineConfiguraiton config, Map<String, PipelineArtifact> artifacts) {
221         var steps = new ArrayList<>(artifacts.size());
222         var runOrder = 0;
223         for (var id : new TreeSet<>(artifacts.keySet())) {
224             var artifact = artifacts.get(id);
225             var type = ActionTypeId.s3Deploy();
226             var actionConfig = new HashMap<String, Object>();
227             actionConfig.put("BucketName", artifact.getNextBucketName());
228             actionConfig.put("ObjectKey", artifact.getObjectKey());
229             actionConfig.put("Extract", false);
230             // this shifts ownership to target account in case (usual scenario) of cross-account deployment
231             actionConfig.put("CannedACL", "bucket-owner-full-control");
232             steps.add(ProcessedTemplate.buildActionDefinition(id, type, actionConfig, Collections.singletonList(id), Collections.emptyList(), ++runOrder));
233         }
234         return ProcessedTemplate.fnIf(config.getHasNextStageConditionName(), CodePipelineUtils.buildStage("Promote", steps), ProcessedTemplate.NO_VALUE);
235     }

262     /**
263      * Orders all actions.
264      *
265      * @param actions Stage actions.
266      */
267     private static void orderActions(List<PipelineAction> actions) {
268         var visited = new HashSet<PipelineAction>();
269         var outputs = new HashMap<String, PipelineAction>();
270         actions.forEach(action -> action.getOutputs().forEach(input -> outputs.put(input, action)));
271         actions.forEach(action -> ProcessedTemplate.calculateActionOrder(action, visited, outputs));
272     }

274     /**
275      * Calculates action order.
276      *
277      * @param action Current subject action.
278      * @param visited Current state markers.
279      * @param outputs Output artifacts mapping.
280      */
281     private static void calculateActionOrder(PipelineAction action, Set<PipelineAction> visited, Map<String, PipelineAction> outputs) {
282         // already calculated
283         if (action.getRunOrder() != null) {
284             return;
285         }
286         if (visited.contains(action)) {
287             throw new IllegalArgumentException(String.format("Circular artifact dependency for %s.", action.getName()));
288         }
289         visited.add(action);
290         
291         // dependencies from outside of stage not need to be considered
292         action.getInputs().stream().filter(outputs::containsKey).peek(input -> ProcessedTemplate.calculateActionOrder(outputs.get(input), visited, outputs)).map(input -> outputs.get(input).getRunOrder()).mapToInt(order -> (order == null ? 1 : order) + 1).max().ifPresent(action::setRunOrder);
293         visited.remove(action);
294     }

307      * Builds list of all used input artifacts.
308      *
309      * @param action Action definition.
310      * @return List of input artifacts.
311      */
312     private static List<String> detectInputArtifacts(PipelineAction action) {
313         var inputs = action.getInputs();
314         // check TemplatePath source artifact
315         var config = action.getConfiguration();
316         var value = config.get("TemplatePath");
317         if (value instanceof String) {
318             inputs.add(((String) value).split(ProcessedTemplate.ARTIFACT_SEPARATOR)[0]);
319         }
320         // check TemplateConfiguration source artifact
321         value = config.get("TemplateConfiguration");
322         if (value instanceof String) {
323             inputs.add(((String) value).split(ProcessedTemplate.ARTIFACT_SEPARATOR)[0]);
324         }
325         return inputs;
326     }

343     /**
344      * Populates pipeline action definition structure.
345      *
346      * @param action Action configuration.
347      * @param actionName Action name.
348      * @param type Action type.
349      * @param configuration Custom options.
350      * @param inputs Input artifacts list.
351      * @param outputs Output artifacts list.
352      * @param runOrder Execution order.
353      * @return CodePipeline stage definition.
354      */
355     private static Map<String, Object> populateActionDefinition(Map<String, Object> action, String actionName, ActionTypeId type, Map<String, Object> configuration, List<String> inputs, List<String> outputs, Integer runOrder) {
356         action.put("Name", actionName);
357         action.compute("ActionTypeId", (String key, Object old) -> CodePipelineUtils.convertActionTypeId(type != null ? type : old));
358         if (!configuration.isEmpty()) {
359             action.put("Configuration", configuration);
360         }
361         if (!inputs.isEmpty()) {
362             action.put("InputArtifacts", inputs.stream().distinct().map(CodePipelineUtils::buildArtifactRef).collect(Collectors.toList()));
363         }
364         if (!outputs.isEmpty()) {
365             action.put("OutputArtifacts", outputs.stream().distinct().map(CodePipelineUtils::buildArtifactRef).collect(Collectors.toList()));
366         }
367         if (runOrder != null) {
368             action.put("RunOrder", runOrder);
369         }
370         return action;
371     }

374      * Builds webhook definition.
375      *
376      * @param config Pipeline configuration.
377      * @param checkoutAction Checkout action name.
378      * @return Resource definition.
379      */
380     private Map<String, Object> buildWebhookDefinition(PipelineConfiguraiton config, String checkoutAction) {
381         var filter = new HashMap<String, String>();
382         filter.put("JsonPath", "$.ref");
383         filter.put("MatchEquals", "refs/heads/{Branch}");
384         var properties = new HashMap<String, Object>();
385         properties.put("Authentication", config.getWebhookAuthenticationType());
386         properties.put("AuthenticationConfiguration", Collections.singletonMap("SecretToken", config.getWebhookSecretToken()));
387         properties.put("TargetPipeline", TemplateUtils.ref(config.getResourceName()));
388         properties.put("TargetPipelineVersion", TemplateUtils.getAtt(config.getResourceName(), "Version"));
389         properties.put("TargetAction", checkoutAction);
390         properties.put("Filters", Collections.singletonList(filter));
391         properties.put("RegisterWithThirdParty", true);
392         return TemplateUtils.generateResource("CodePipeline::Webhook", properties, config.getHasCheckoutStepConditionName());
393     }
