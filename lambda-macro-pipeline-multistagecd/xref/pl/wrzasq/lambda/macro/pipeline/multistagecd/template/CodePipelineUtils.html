8   package pl.wrzasq.lambda.macro.pipeline.multistagecd.template;
9   
10  import java.util.Collections;
11  import java.util.HashMap;
12  import java.util.Map;
13  
14  import pl.wrzasq.lambda.macro.pipeline.multistagecd.model.codepipeline.ActionTypeId;
15  
16  /**
17   * CodePipeline utilities.
18   */
19  public class CodePipelineUtils {
20      /**
21       * "Name" property key.
22       */
23      private static final String KEY_NAME = "Name";
24  
25      /**
26       * "Actions" property key.
27       */
28      private static final String KEY_ACTIONS = "Actions";
29  
30      /**
31       * Builds artifact structure.
32       *
33       * @param name Artifact name.
34       * @return CloudFormation structure.
35       */
36      public static Map<String, String> buildArtifactRef(String name) {
37          return Collections.singletonMap(CodePipelineUtils.KEY_NAME, name);
38      }
39  
40      /**
41       * Builds stage structure.
42       *
43       * @param name Stage name.
44       * @param actions Pipeline actions.
45       * @return CloudFormation structure.
46       */
47      public static Map<String, Object> buildStage(String name, Object actions) {
48          var stage = new HashMap<String, Object>();
49          stage.put(CodePipelineUtils.KEY_NAME, name);
50          stage.put(CodePipelineUtils.KEY_ACTIONS, actions);
51          return stage;
52      }
53  
54      /**
55       * Converts type model to plain map.
56       *
57       * @param input CodePipeline model.
58       * @return CloudFormation structure.
59       */
60      public static Object convertActionTypeId(Object input) {
61          if (input instanceof ActionTypeId) {
62              var type = (ActionTypeId) input;
63              var value = new HashMap<String, String>();
64              value.put("Category", type.getCategory());
65              value.put("Owner", type.getOwner());
66              value.put("Provider", type.getProvider());
67              value.put("Version", type.getVersion());
68              return value;
69          } else {
70              return input;
71          }
72      }
73  }
