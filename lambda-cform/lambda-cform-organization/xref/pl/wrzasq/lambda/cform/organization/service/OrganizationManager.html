8   package pl.wrzasq.lambda.cform.organization.service;
9   
10  import com.amazonaws.SdkBaseException;
11  import com.amazonaws.services.organizations.AWSOrganizations;
12  import com.amazonaws.services.organizations.model.CreateOrganizationRequest;
13  import com.amazonaws.services.organizations.model.CreateOrganizationResult;
14  import com.amazonaws.services.organizations.model.DeleteOrganizationRequest;
15  import com.amazonaws.services.organizations.model.DescribeOrganizationRequest;
16  import com.amazonaws.services.organizations.model.DescribeOrganizationResult;
17  import com.amazonaws.services.organizations.model.Organization;
18  import org.slf4j.Logger;
19  import org.slf4j.LoggerFactory;
20  import pl.wrzasq.commons.aws.cloudformation.CustomResourceResponse;
21  import pl.wrzasq.lambda.cform.organization.model.OrganizationRequest;
22  
23  /**
24   * Organizations API implementation.
25   */
26  public class OrganizationManager
27  {
28      /**
29       * Logger.
30       */
31      private Logger logger = LoggerFactory.getLogger(OrganizationManager.class);
32  
33      /**
34       * AWS Organizations API client.
35       */
36      private AWSOrganizations organizations;
37  
38      /**
39       * Initializes object with given Organizations client.
40       *
41       * @param organizations AWS Organizations client.
42       */
43      public OrganizationManager(AWSOrganizations organizations)
44      {
45          this.organizations = organizations;
46      }
47  
48      /**
49       * Handles organization creation.
50       *
51       * @param input Resource creation request.
52       * @return Data about published version.
53       */
54      public CustomResourceResponse<Organization> sync(OrganizationRequest input)
55      {
56          try {
57              DescribeOrganizationResult result = this.organizations.describeOrganization(
58                  new DescribeOrganizationRequest()
59              );
60  
61              this.logger.info("Organization already exists (ID {}).", result.getOrganization().getId());
62  
63              return new CustomResourceResponse<>(result.getOrganization(), result.getOrganization().getArn());
64          } catch (SdkBaseException error) {
65              this.logger.info("Exception occurred during organization data fetching, probably doesn't exist.", error);
66  
67              CreateOrganizationResult result = this.organizations.createOrganization(
68                  new CreateOrganizationRequest()
69                      .withFeatureSet(input.getFeatureSet())
70              );
71  
72              this.logger.info("Created new organization, ID {}.", result.getOrganization().getId());
73  
74              return new CustomResourceResponse<>(result.getOrganization(), result.getOrganization().getArn());
75          }
76      }
77  
78      /**
79       * Handles organization deletion.
80       *
81       * @param input Resource delete request.
82       * @return Data about deleted unit.
83       */
84      public CustomResourceResponse<Organization> delete(OrganizationRequest input)
85      {
86          this.organizations.deleteOrganization(new DeleteOrganizationRequest());
87  
88          this.logger.info("Organization deleted.");
89  
90          return new CustomResourceResponse<>(null);
91      }
92  }
