63      /**
64       * Handles account creation.
65       *
66       * @param input Resource creation request.
67       * @param physicalResourceId Physical ID of existing resource (in this case always null).
68       * @return Data about published version.
69       */
70      public CustomResourceResponse<Account> provision(AccountRequest input, String physicalResourceId) {
71          if (physicalResourceId != null) {
72              physicalResourceId = resolveExistingAccount(physicalResourceId, input);
73          }
74          // new account needed
75          if (physicalResourceId == null) {
76              physicalResourceId = this.initializeAccount(input);
77          }
78          // check current location in organization structure
79          Parent parent = this.organizations.listParents(new ListParentsRequest().withChildId(physicalResourceId)).getParents().get(0);
80          if (!parent.getId().equals(input.getOuId())) {
81              this.logger.info("Moving account {} from {} to {}.", physicalResourceId, parent.getId(), input.getOuId());
82              this.organizations.moveAccount(new MoveAccountRequest().withAccountId(physicalResourceId).withSourceParentId(parent.getId()).withDestinationParentId(input.getOuId()));
83          }
84          Account account = this.organizations.describeAccount(new DescribeAccountRequest().withAccountId(physicalResourceId)).getAccount();
85          return new CustomResourceResponse<>(account, physicalResourceId);
86      }

111     /**
112      * Creates plain, new account.
113      *
114      * @param email E-mail address for root access.
115      * @param name Account label.
116      * @param roleName Administration role name.
117      * @return Account ID.
118      */
119     private String createAccount(String email, String name, String roleName) {
120         CreateAccountStatus status = this.organizations.createAccount(new CreateAccountRequest().withEmail(email).withAccountName(name).withRoleName(roleName)).getCreateAccountStatus();
121         this.logger.info("New account creation initialized for {} ({}).", email, name);
122         // wait until account creation process is finished
123         while (CreateAccountState.fromValue(status.getState()) == CreateAccountState.IN_PROGRESS) {
124             this.logger.info("Account creation {} in progress…", name);
125             this.sleep();
126             status = this.organizations.describeCreateAccountStatus(new DescribeCreateAccountStatusRequest().withCreateAccountRequestId(status.getId())).getCreateAccountStatus();
127         }
128         this.logger.info("Account creation status: {}.", status.getState());
129         if (CreateAccountState.fromValue(status.getState()) == CreateAccountState.SUCCEEDED) {
130             return status.getAccountId();
131         } else {
132             throw new IllegalStateException(String.format("Failed to create account for %s (%s) - reason: %s.", email, name, status.getFailureReason()));
133         }
134     }

136     /**
137      * Invites existing account to organization.
138      *
139      * @param accountId Existing account ID.
140      * @return Account ID.
141      */
142     private String inviteAccount(String accountId) {
143         Handshake handshake = this.organizations.inviteAccountToOrganization(new InviteAccountToOrganizationRequest().withTarget(new HandshakeParty().withType(HandshakePartyType.ACCOUNT).withId(accountId))).getHandshake();
144         this.logger.info("Invited account {} to the organization", accountId);
145         while (HandshakeState.fromValue(handshake.getState()) == HandshakeState.REQUESTED) {
146             this.logger.info("Account {} handshake in progress…", accountId);
147             this.sleep();
148             handshake = this.organizations.describeHandshake(new DescribeHandshakeRequest().withHandshakeId(handshake.getId())).getHandshake();
149         }
150         if (HandshakeState.fromValue(handshake.getState()) == HandshakeState.ACCEPTED) {
151             return accountId;
152         } else {
153             throw new IllegalStateException(String.format("Failed to invite account %s - status: %s.", accountId, handshake.getState()));
154         }
155     }

157     /**
158      * Resolves ID of existing resource.
159      *
160      * @param accountId Currently provisioned ID.
161      * @param input Desired account specification.
162      * @return Resolved account ID.
163      */
164     private String resolveExistingAccount(String accountId, AccountRequest input) {
165         try {
166             Account account = this.organizations.describeAccount(new DescribeAccountRequest().withAccountId(accountId)).getAccount();
167             if (input.getEmail().equals(account.getEmail()) && input.getAccountName().equals(account.getName())) {
168                 return account.getId();
169             } else {
170                 this.logger.warn("Account {} core data changed - will re-create account!", account.getArn());
171                 return null;
172             }
173         } catch (AccountNotFoundException error) {
174             this.logger.warn("Account {} not found - resolving to new account.", accountId);
175             return null;
176         }
177     }
