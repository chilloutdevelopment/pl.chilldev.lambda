8   package pl.wrzasq.lambda.cform.cognito.domain.data.service;
9   
10  import com.amazonaws.services.cognitoidp.AWSCognitoIdentityProvider;
11  import com.amazonaws.services.cognitoidp.model.DescribeUserPoolDomainRequest;
12  import com.amazonaws.services.cognitoidp.model.DomainDescriptionType;
13  import pl.wrzasq.commons.aws.cloudformation.CustomResourceResponse;
14  import pl.wrzasq.lambda.cform.cognito.domain.data.model.CognitoDomainDataRequest;
15  
16  /**
17   * Cognito API implementation.
18   */
19  public class CognitoDomainHandler {
20      /**
21       * AWS Cognito API client.
22       */
23      private AWSCognitoIdentityProvider cognitoIdp;
24  
25      /**
26       * Initializes object with given Cognito client.
27       *
28       * @param cognitoIdp AWS Cognito client.
29       */
30      public CognitoDomainHandler(AWSCognitoIdentityProvider cognitoIdp) {
31          this.cognitoIdp = cognitoIdp;
32      }
33  
34      /**
35       * Handles domain description.
36       *
37       * @param input Resource creation request.
38       * @param physicalResourceId Physical ID of existing resource (if present).
39       * @return Data about created project.
40       */
41      public CustomResourceResponse<DomainDescriptionType> read(
42          CognitoDomainDataRequest input,
43          String physicalResourceId
44      ) {
45          var data = this.cognitoIdp.describeUserPoolDomain(
46              new DescribeUserPoolDomainRequest()
47                  .withDomain(input.getDomain())
48          )
49              .getDomainDescription();
50  
51          return new CustomResourceResponse<>(
52              data,
53              input.getDomain()
54          );
55      }
56  
57      /**
58       * Handles project deletion.
59       *
60       * @param input Resource delete request.
61       * @param physicalResourceId Physical ID of existing resource (if present).
62       * @return Empty response.
63       */
64      public CustomResourceResponse<DomainDescriptionType> delete(
65          CognitoDomainDataRequest input,
66          String physicalResourceId
67      ) {
68          return new CustomResourceResponse<>(null, physicalResourceId);
69      }
70  }
