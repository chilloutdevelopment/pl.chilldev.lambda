37       * Message pattern for drift case.
38       */
39      private static final String DRIFT_LOG_MESSAGE_PATTERN
40          = "Stack set ID {} differs from CloudFormation-provided physical resource ID {}.";

42      /**
43       * Cast array.
44       */
45      private static final Capability[] CAPABILITY_CAST = new Capability[0];

62      /**
63       * Initializes object with given CloudFormation client.
64       *
65       * @param cloudFormation AWS CloudFormation client.
66       * @param stackSetHandler Stack set operations helper.
67       */
68      public StackSetManager(AmazonCloudFormation cloudFormation, StackSetHandler stackSetHandler) {
69          this.cloudFormation = cloudFormation;
70          this.stackSetHandler = stackSetHandler;
71      }

74       * Handles stack set deployment.
75       *
76       * @param input Resource deployment request.
77       * @param physicalResourceId Physical ID of existing resource (if present).
78       * @return Data about published version.
79       */
80      public CustomResourceResponse<StackSetResponse> deployStackSet(StackSetRequest input, String physicalResourceId) {

91              this.updateStackSet(input);

93              // don't do anything here - in worst case it will fail in Delete call in UPDATE_COMPLETE_CLEANUP phase

96              if (!stackSet.getStackSetId().equals(physicalResourceId)) {
97                  this.logger.warn(

102             }

104             response.setId(stackSet.getStackSetId());

106             response.setId(this.createStackSet(input));
107         }
108 
109         response.setStackSetName(input.getStackSetName());
110         return new CustomResourceResponse<>(response, response.getId());
111     }

113     /**
114      * Handles stack set deletion.
115      *
116      * @param input Resource delete request.
117      * @param physicalResourceId Physical ID of existing resource (if present).
118      * @return Empty response.
119      */
120     public CustomResourceResponse<StackSetResponse> deleteStackSet(StackSetRequest input, String physicalResourceId) {
121         StackSet stackSet = this.cloudFormation.describeStackSet(
122             new DescribeStackSetRequest()
123                 .withStackSetName(input.getStackSetName())
124         )
125             .getStackSet();
126 
127         // avoid removing unknown data
128         if (!stackSet.getStackSetId().equals(physicalResourceId)) {
129             this.logger.error(
130                 StackSetManager.DRIFT_LOG_MESSAGE_PATTERN,
131                 stackSet.getStackSetId(),
132                 physicalResourceId
133             );
134             throw new IllegalStateException(
135                 String.format(
136                     "Can not delete Stack set - ID %s doesn't match CloudFormation-provided resource ID %s.",
137                     stackSet.getStackSetId(),
138                     physicalResourceId
139                 )
140             );
141         }
142 
143         this.cloudFormation.deleteStackSet(
144             new DeleteStackSetRequest()
145                 .withStackSetName(input.getStackSetName())
146         );
147 
148         this.logger.info("Stack set deleted.");
149 
150         return new CustomResourceResponse<>(null, physicalResourceId);
151     }
152 
153     /**
154      * Creates new stack set.
155      *
156      * @param input Stack set specification.
157      * @return Created stack set ID.
158      */
159     private String createStackSet(StackSetRequest input) {
160         CreateStackSetResult result = this.cloudFormation.createStackSet(
161             new CreateStackSetRequest()
162                 .withStackSetName(input.getStackSetName())
163                 .withTemplateURL(input.getTemplateUrl())
164                 .withDescription(input.getDescription())
165                 .withAdministrationRoleARN(input.getAdministrationRoleArn())
166                 .withExecutionRoleName(input.getExecutionRoleName())
167                 .withCapabilities(input.getCapabilities().toArray(StackSetManager.CAPABILITY_CAST))
168                 .withParameters(StackSetManager.buildSdkParameters(input))
169                 .withTags(StackSetManager.buildSdkTags(input))
170         );
171 
172         this.logger.info("Created new stack set, ID {}.", result.getStackSetId());
173 
174         return result.getStackSetId();
175     }
176 
177     /**
178      * Updates existing stack set.
179      *
180      * @param input Stack set specification.
181      */
182     private void updateStackSet(StackSetRequest input) {
183         UpdateStackSetResult result = this.cloudFormation.updateStackSet(
184             new UpdateStackSetRequest()
185                 .withStackSetName(input.getStackSetName())
186                 .withTemplateURL(input.getTemplateUrl())
187                 .withDescription(input.getDescription())
188                 .withAdministrationRoleARN(input.getAdministrationRoleArn())
189                 .withExecutionRoleName(input.getExecutionRoleName())
190                 .withCapabilities(input.getCapabilities().toArray(StackSetManager.CAPABILITY_CAST))
191                 .withParameters(StackSetManager.buildSdkParameters(input))
192                 .withTags(StackSetManager.buildSdkTags(input))
193         );
194 
195         this.stackSetHandler.waitForStackSetOperation(input.getStackSetName(), result.getOperationId());
196 
197         this.logger.info("Updated stack set (operation ID: {}).", result.getOperationId());
198     }

200     /**
201      * Converts key-value mapping into AWS SDK structure.
202      *
203      * @param input Request data.
204      * @return Collection of AWS SDK DTOs.
205      */
206     private static Collection<Parameter> buildSdkParameters(StackSetRequest input) {
207         return StackUtils.buildSdkList(
208             input.getParameters(),
209             (String key, String value) ->
210                 new Parameter()
211                     .withParameterKey(key)
212                     .withParameterValue(value)
213         );
214     }

222     private static Collection<Tag> buildSdkTags(StackSetRequest input) {
223         return StackUtils.buildSdkList(
224             input.getTags(),
225             (String key, String value) ->
226                 new Tag()
227                     .withKey(key)
228                     .withValue(value)
229         );
230     }
