29  /**
30   * CloudFormation API implementation.
31   */
32  public class StackSetManager {
33      /**
34       * Message pattern for drift case.
35       */
36      private static final String DRIFT_LOG_MESSAGE_PATTERN
37          = "Stack set ID {} differs from CloudFormation-provided physical resource ID {}.";
38  
39      /**
40       * Cast array.
41       */
42      private static final Capability[] CAPABILITY_CAST = new Capability[0];
43  
44      /**
45       * Logger.
46       */
47      private Logger logger = LoggerFactory.getLogger(StackSetManager.class);
48  
49      /**
50       * AWS CloudFormation API client.
51       */
52      private AmazonCloudFormation cloudFormation;
53  
54      /**
55       * Stack set operations helper.
56       */
57      private StackSetHandler stackSetHandler;
58  
59      /**
60       * Initializes object with given CloudFormation client.
61       *
62       * @param cloudFormation AWS CloudFormation client.
63       * @param stackSetHandler Stack set operations helper.
64       */
65      public StackSetManager(AmazonCloudFormation cloudFormation, StackSetHandler stackSetHandler) {
66          this.cloudFormation = cloudFormation;
67          this.stackSetHandler = stackSetHandler;
68      }
69  
70      /**
71       * Handles stack set deployment.
72       *
73       * @param input Resource deployment request.
74       * @param physicalResourceId Physical ID of existing resource (if present).
75       * @return Data about published version.
76       */
77      public CustomResourceResponse<StackSetResponse> deployStackSet(StackSetRequest input, String physicalResourceId) {
78          var response = new StackSetResponse();
79  
80          try {
81              var stackSet = this.cloudFormation.describeStackSet(
82                  new DescribeStackSetRequest()
83                      .withStackSetName(input.getStackSetName())
84              )
85                  .getStackSet();
86              this.logger.info("Stack set already exists (ARN {}).", stackSet.getStackSetARN());
87  
88              this.updateStackSet(input);
89  
90              // don't do anything here - in worst case it will fail in Delete call in UPDATE_COMPLETE_CLEANUP phase
91              // it will not hurt us (even if fails, as it's post-update phase) and allows to simplify logic of this
92              // method to avoid handling all combinations, which could lead to unhandled edge cases
93              if (!stackSet.getStackSetId().equals(physicalResourceId)) {
94                  this.logger.warn(
95                      StackSetManager.DRIFT_LOG_MESSAGE_PATTERN,
96                      stackSet.getStackSetId(),
97                      physicalResourceId
98                  );
99              }
100 
101             response.setId(stackSet.getStackSetId());
102         } catch (StackSetNotFoundException error) {
103             response.setId(this.createStackSet(input));
104         }
105 
106         response.setStackSetName(input.getStackSetName());
107         return new CustomResourceResponse<>(response, response.getId());
108     }
109 
110     /**
111      * Handles stack set deletion.
112      *
113      * @param input Resource delete request.
114      * @param physicalResourceId Physical ID of existing resource (if present).
115      * @return Empty response.
116      */
117     public CustomResourceResponse<StackSetResponse> deleteStackSet(StackSetRequest input, String physicalResourceId) {
118         var stackSet = this.cloudFormation.describeStackSet(
119             new DescribeStackSetRequest()
120                 .withStackSetName(input.getStackSetName())
121         )
122             .getStackSet();
123 
124         // avoid removing unknown data
125         if (!stackSet.getStackSetId().equals(physicalResourceId)) {
126             this.logger.error(
127                 StackSetManager.DRIFT_LOG_MESSAGE_PATTERN,
128                 stackSet.getStackSetId(),
129                 physicalResourceId
130             );
131             throw new IllegalStateException(
132                 String.format(
133                     "Can not delete Stack set - ID %s doesn't match CloudFormation-provided resource ID %s.",
134                     stackSet.getStackSetId(),
135                     physicalResourceId
136                 )
137             );
138         }
139 
140         this.cloudFormation.deleteStackSet(
141             new DeleteStackSetRequest()
142                 .withStackSetName(input.getStackSetName())
143         );
144 
145         this.logger.info("Stack set deleted.");
146 
147         return new CustomResourceResponse<>(null, physicalResourceId);
148     }
149 
150     /**
151      * Creates new stack set.
152      *
153      * @param input Stack set specification.
154      * @return Created stack set ID.
155      */
156     private String createStackSet(StackSetRequest input) {
157         var result = this.cloudFormation.createStackSet(
158             new CreateStackSetRequest()
159                 .withStackSetName(input.getStackSetName())
160                 .withTemplateURL(input.getTemplateUrl())
161                 .withDescription(input.getDescription())
162                 .withAdministrationRoleARN(input.getAdministrationRoleArn())
163                 .withExecutionRoleName(input.getExecutionRoleName())
164                 .withCapabilities(input.getCapabilities().toArray(StackSetManager.CAPABILITY_CAST))
165                 .withParameters(StackSetManager.buildSdkParameters(input))
166                 .withTags(StackSetManager.buildSdkTags(input))
167         );
168 
169         this.logger.info("Created new stack set, ID {}.", result.getStackSetId());
170 
171         return result.getStackSetId();
172     }
173 
174     /**
175      * Updates existing stack set.
176      *
177      * @param input Stack set specification.
178      */
179     private void updateStackSet(StackSetRequest input) {
180         var result = this.cloudFormation.updateStackSet(
181             new UpdateStackSetRequest()
182                 .withStackSetName(input.getStackSetName())
183                 .withTemplateURL(input.getTemplateUrl())
184                 .withDescription(input.getDescription())
185                 .withAdministrationRoleARN(input.getAdministrationRoleArn())
186                 .withExecutionRoleName(input.getExecutionRoleName())
187                 .withCapabilities(input.getCapabilities().toArray(StackSetManager.CAPABILITY_CAST))
188                 .withParameters(StackSetManager.buildSdkParameters(input))
189                 .withTags(StackSetManager.buildSdkTags(input))
190         );
191 
192         this.stackSetHandler.waitForStackSetOperation(input.getStackSetName(), result.getOperationId());
193 
194         this.logger.info("Updated stack set (operation ID: {}).", result.getOperationId());
195     }
196 
197     /**
198      * Converts key-value mapping into AWS SDK structure.
199      *
200      * @param input Request data.
201      * @return Collection of AWS SDK DTOs.
202      */
203     private static Collection<Parameter> buildSdkParameters(StackSetRequest input) {
204         return StackUtils.buildSdkList(
205             input.getParameters(),
206             (key, value) ->
207                 new Parameter()
208                     .withParameterKey(key)
209                     .withParameterValue(value)
210         );
211     }
212 
213     /**
214      * Converts key-value mapping into AWS SDK structure.
215      *
216      * @param input Request data.
217      * @return Collection of AWS SDK DTOs.
218      */
219     private static Collection<Tag> buildSdkTags(StackSetRequest input) {
220         return StackUtils.buildSdkList(
221             input.getTags(),
222             (key, value) ->
223                 new Tag()
224                     .withKey(key)
225                     .withValue(value)
226         );
227     }
